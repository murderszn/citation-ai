            <label for="assets">Total Assets:
                <span class="tooltip">?
                    <span class="tooltiptext">Assets include cash, property, investments, vehicles, etc.</span>
                </span>
            </label>
            <input type="number" id="assets" name="assets" placeholder="Enter total assets" required aria-required="true" min="0">

            <label for="liabilities">Total Liabilities:
                <span class="tooltip">?
                    <span class="tooltiptext">Liabilities include loans, mortgages, credit card debt, etc.</span>
                </span>
            </label>
            <input type="number" id="liabilities" name="liabilities" placeholder="Enter total liabilities" required aria-required="true" min="0">

            <button type="button" onclick="calculateNetWorth()">Calculate Net Worth</button>
        </form>

        <div class="result" id="result"></div>

    <script>
        function calculateNetWorth() {
            const assets = parseFloat(document.getElementById('assets').value) || 0;
            const liabilities = parseFloat(document.getElementById('liabilities').value) || 0;
            const netWorth = assets - liabilities;
            document.getElementById('result').textContent = `Your Net Worth is $${netWorth.toLocaleString()}`;
        }
    </script>
